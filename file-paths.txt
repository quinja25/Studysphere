    const SUBJECT_FILE = 'subject-names.txt';
    const SESSION_FILE = 'exam-sessions.txt';
    const PATH_FILE = 'file-paths.txt';
    const SERVER_PREFIX = '/IB%20PAST%20PAPERS%20-%20YEAR/';

    const statusEl = document.getElementById('status');

    function setStatus(msg) {
      statusEl.textContent = msg;
    }

    async function loadList(fileName, containerId) {
      const res = await fetch(fileName);
      if (!res.ok) throw new Error(`Failed to load ${fileName}`);

      const text = await res.text();
      const items = text
        .split(/\r?\n/)
        .map(x => x.trim())
        .filter(Boolean);

      const container = document.getElementById(containerId);
      container.innerHTML = '';

      items.forEach((item, i) => {
        const id = `${containerId}-${i}`;
        const label = document.createElement('label');
        label.innerHTML = `<input type="checkbox" value="${escapeHtml(item)}" id="${id}"> ${escapeHtml(item)}`;
        container.appendChild(label);
      });
    }

    function escapeHtml(str) {
      return str
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;')
        .replace(/'/g, '&#039;');
    }

    function getCheckedValues(containerId) {
      return [...document.querySelectorAll(`#${containerId} input:checked`)]
        .map(cb => cb.value);
    }

    function buildDownloadUrl(originalPath) {
      if (!originalPath.startsWith('./')) {
        throw new Error(`Unexpected path format: ${originalPath}`);
      }

      const relativePath = originalPath.slice(2); // remove './'
      const encodedPath = relativePath
        .split('/')
        .map(part => encodeURIComponent(part))
        .join('/');

      return SERVER_PREFIX + encodedPath;
    }

    async function ensureDirectory(rootHandle, pathParts) {
      let current = rootHandle;
      for (const part of pathParts) {
        current = await current.getDirectoryHandle(part, { create: true });
      }
      return current;
    }

    async function saveFile(rootHandle, relativePath, blob) {
      const parts = relativePath.split('/');
      const fileName = parts.pop();
      const dirHandle = await ensureDirectory(rootHandle, parts);

      const fileHandle = await dirHandle.getFileHandle(fileName, { create: true });
      const writable = await fileHandle.createWritable();
      await writable.write(blob);
      await writable.close();
    }

    async function downloadMatchingFiles() {
      const selectedSubjects = getCheckedValues('subjects');
      const selectedSessions = getCheckedValues('sessions');

      if (!selectedSubjects.length || !selectedSessions.length) {
        alert('Please select at least one subject and one exam session.');
        return;
      }

      if (!window.showDirectoryPicker) {
        alert('This browser does not support the File System Access API. Use Chromium-based desktop browsers.');
        setStatus('This browser does not support the File System Access API. Use Chromium-based desktop browsers.');
        document.getElementById('downloadBtn').hidden = true;
        return;
      }

      setStatus('Choose a destination folder...');
      const rootHandle = await window.showDirectoryPicker();

      setStatus('Loading file paths...');
      const pathRes = await fetch(PATH_FILE);
      if (!pathRes.ok) throw new Error('Failed to load file-paths.txt');

      const allPaths = (await pathRes.text())
        .split(/\r?\n/)
        .map(x => x.trim())
        .filter(Boolean);

      const matching = allPaths.filter(path => {
        const subjectMatch = selectedSubjects.some(s => path.includes(s));
        const sessionMatch = selectedSessions.some(s => path.includes(s));
        return subjectMatch && sessionMatch;
      });

      if (!matching.length) {
        setStatus('No matching files found.');
        document.getElementById('downloadBtn').hidden = false;
        return;
      }

      setStatus(`Found ${matching.length} files. Starting download...`);

      let completed = 0;
      let failed = 0;

      for (const originalPath of matching) {
        try {
          const relativePath = originalPath.slice(2); // preserve original structure
          const url = buildDownloadUrl(originalPath);

          setStatus(`Downloading ${completed + failed + 1}/${matching.length}\n${relativePath}`);

          const res = await fetch(url);
          if (!res.ok) throw new Error(`HTTP ${res.status}`);

          const blob = await res.blob();
          await saveFile(rootHandle, relativePath, blob);
          completed++;
        } catch (err) {
          console.error('Failed:', originalPath, err);
          failed++;
        }
      }

      setStatus(
        `Done.\n` +
        `Downloaded: ${completed}\n` +
        `Failed: ${failed}`
      );
    }

    document.getElementById('downloadBtn').addEventListener('click', async () => {
      try {
        document.getElementById('downloadBtn').hidden = true;
        await downloadMatchingFiles();
      } catch (err) {
        console.error(err);
        setStatus(`Error: ${err.message}. Reload this page to try again.`);
      }
    });

    (async function init() {
      try {
        await Promise.all([
          loadList(SUBJECT_FILE, 'subjects'),
          loadList(SESSION_FILE, 'sessions')
        ]);
        setStatus('Ready. Select filters and click download.');
        if (!window.showDirectoryPicker) {
          setStatus('This browser does not support the File System Access API. Use Chromium-based desktop browsers.');
          document.getElementById('downloadBtn').hidden = true;
          return;
        }
      } catch (err) {
        console.error(err);
        setStatus(`Initialization error: ${err.message}. Reload this page to try again.`);
      }
    })();
